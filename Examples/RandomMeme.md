# Example For RandomMeme

```js
const { RandomMeme } = require('klaymon')


await RandomMeme({
    message: message, // Message Parameter
    embed: { 
        color: "PURPLE", //Embed color / DEFAULT: {RANDOM} VALUE: {STRING}
        timestamp: true, //Embed Timestamp / DEFAULT: {TRUE} VALUE: {BOOLEAN}
        
    },
    
})```
